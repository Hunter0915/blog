SELECT CITY, LENGTH(CITY) AS city_length
FROM STATION
WHERE LENGTH(CITY) = (SELECT MIN(LENGTH(CITY)) FROM STATION)
ORDER BY CITY
LIMIT 1

UNION ALL

SELECT CITY, LENGTH(CITY) AS city_length
FROM STATION
WHERE LENGTH(CITY) = (SELECT MAX(LENGTH(CITY)) FROM STATION)
ORDER BY CITY
LIMIT 1;

SELECT DISTINCT CITY
FROM STATION
WHERE UPPER(LEFT(CITY, 1)) IN ('А', 'Е', 'Ё', 'И', 'О', 'У', 'Ы', 'Э', 'Ю', 'Я');

SELECT CONCAT(Name, '(', LEFT(Occupation, 1), ')') AS formatted_name
FROM OCCUPATIONS
ORDER BY Name;

SELECT 
    MAX(CASE WHEN Occupation = 'Ученик' THEN Name END) AS Ученик,
    MAX(CASE WHEN Occupation = 'Студент' THEN Name END) AS Студент,
    MAX(CASE WHEN Occupation = 'Преподаватель' THEN Name END) AS Преподаватель
FROM (
    SELECT 
        Name,
        Occupation,
        ROW_NUMBER() OVER (PARTITION BY Occupation ORDER BY Name) AS rn
    FROM OCCUPATIONS
) t
GROUP BY rn
ORDER BY rn;

SELECT 
    ROUND(ABS(MAX(LAT_N) - MIN(LAT_N)) + ABS(MAX(LONG_W) - MIN(LONG_W)), 4) AS manhattan_distance
FROM STATION;

SELECT 
    ROUND(SQRT(POWER(MAX(LAT_N) - MIN(LAT_N), 2) + POWER(MAX(LONG_W) - MIN(LONG_W), 2)), 4) AS euclidean_distance
FROM STATION;


# Базовый класс
class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        raise NotImplementedError("Должен быть реализован в дочернем классе")

    def move(self):
        print(f"{self.name} двигается")

# Дочерние классы
class Dog(Animal):
    def speak(self):
        return f"{self.name} говорит: Гав!"

    def fetch(self):
        print(f"{self.name} приносит палку")

class Cat(Animal):
    def speak(self):
        return f"{self.name} говорит: Мяу!"

    def climb(self):
        print(f"{self.name} залезает на дерево")

# Использование
animals = [
    Dog("Барсик"),
    Cat("Мурка")
]

for animal in animals:
    print(animal.speak())
    animal.move()

# Уникальные методы каждого класса
dog = Dog("Рекс")
cat = Cat("Василий")
dog.fetch()
cat.climb()


# Разные классы с одинаковым методом
class Rectangle:
    def __init__(self, width, height):
        self.width = width
        self.height = height

    def area(self):
        return self.width * self.height

class Circle:
    def __init__(self, radius):
        self.radius = radius

    def area(self):
        return 3.14 * self.radius ** 2

class Triangle:
    def __init__(self, base, height):
        self.base = base
        self.height = height

    def area(self):
        return 0.5 * self.base * self.height

# Функция, использующая полиморфизм
def print_area(shape):
    print(f"Площадь фигуры: {shape.area()}")

# Создание объектов разных классов
shapes = [
    Rectangle(5, 10),
    Circle(7),
    Triangle(6, 8)
]

# Одинаковый вызов метода для разных объектов
for shape in shapes:
    print_area(shape)

# Другой пример полиморфизма с оператором +

print([1, 2] + [3, 4]) # объединение списков
